function clock_mistake_plot(method, coords, maxiter)
    res = [[], []]
    dm = 0.01
    mistake = 0.0
    for i in 1:maxiter
        alg_coords = method(coords, mistake)
        dist = MAXdist(coords, alg_coords)
        push!(res[1], mistake)
        push!(res[2], dist)
        mistake+=dm
    end
    res
end;

function iter_mistake_plot(method, coords, maxiter)
    res = [[], []]
    for i in 1:maxiter
        newton_coords = method(coords, 0.005, i)
        dist = MAXdist(coords, newton_coords)
        push!(res[1], i)
        push!(res[2], dist)
    end
    return res
end

function leastSquares_sat_plot(positers=100, satiters=50)
    res = [[],[]]
    
    sat = [ rand_position(sat_rad) for i in 1:satiters ]
    means1 = zeros(satiters)
    iters_ok1 = zeros(satiters)
    means2 = zeros(satiters)
    iters_ok2 = zeros(satiters)
    for i in 1:positers
        x = rand_position(earth_rad)
        prepsat!(x, sat, true)
        for sat_cnt in 4:satiters
            x1 = leastSquares(sat, 10, sat_cnt)[1:3]
            x2 = bancroft(sat)[1:3]
            dx1 = sqrt(sum(abs2, x1-x[1:3]))
            dx2 = sqrt(sum(abs2, x2- x[1:3]))
            if dx1 < 1000
                means1[sat_cnt] += dx1
                iters_ok1[sat_cnt] += 1
            end
            if dx2 < 1000
                means2[sat_cnt] += dx2
                iters_ok2[sat_cnt] += 1
            end
        end
    end
    res1 =[[], []]
    res2 = [[], []]
    for i in 6:satiters
        push!(res1[1], i)
        push!(res1[2], means1[i]/iters_ok1[i])
        push!(res2[1], i)
        push!(res2[2], means2[i]/iters_ok2[i])
    end
    [res1, res2]
end;
